import nltk

7. Shallow Parsing (Chunking)

import nltk
from nltk.tokenize import word_tokenize
from nltk.tag import pos_tag

nltk.download('punkt_tab')
nltk.download('averaged_perceptron_tagger_eng')

words=word_tokenize(text)
print(words)

tagged_words=pos_tag(words)
print(tagged_words)

grammar="NP:{<DT>?<JJ>*<NN>}"

chunk=nltk.RegexpParser(grammar)

result=chunk.parse(tagged_words)
result.pretty_print()

4. Generating Words using NLP

import nltk
import random
import string
from collections import defaultdict

# Download tokenizer
nltk.download('punkt')
nltk.download('punkt_tab')

# Sample text
text = """
Machine learning is useful 
Learning machines are powerful 
Machine learning is important.
"""

# Convert to lowercase
text = text.lower()

# Remove punctuation
text = text.translate(str.maketrans('', '', string.punctuation))

# Split into words
words = nltk.word_tokenize(text)

# Create dictionary
model = defaultdict(list)

# Build bigram model
for i in range(len(words) - 1):
    model[words[i]].append(words[i + 1])

# Function to generate words
def generate_words(start, n):

    result = [start]

    for i in range(n - 1):

        # Stop if no next word exists
        if len(model[result[-1]]) == 0:
            break

        # Choose random next word
        next_word = random.choice(model[result[-1]])

        result.append(next_word)

    return " ".join(result)

# Generate sentence
print(generate_words("machine", 10))
